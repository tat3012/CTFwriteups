from random import randint
from z3 import *

def decrypt(text):
  key = randint(1, 2 * len(text))
  result = []

  for c in text:
    x = Int('x')
    s = Solver()
    s.add(c - (x % key) == x) 
    if s.check() == sat:
      model = s.model()
      result.append(model[x].as_long()) 

    key = key + 1

  return result
